```typescript
import * as vscode from 'vscode';

export function activate(context: vscode.ExtensionContext): void {
  const huntRoot = findHuntRoot();
  if (!huntRoot) {
    return; // Silently exit -- activation event fired but no hunt workspace found
  }

  // Register the info command as proof-of-life
  context.subscriptions.push(
    vscode.commands.registerCommand('thrunt-god.showInfo', () => {
      vscode.window.showInformationMessage(
        `THRUNT God: Hunt workspace detected at ${huntRoot.fsPath}`
      );
    })
  );

  // Log activation for diagnostics
  const outputChannel = vscode.window.createOutputChannel('THRUNT God');
  context.subscriptions.push(outputChannel);
  outputChannel.appendLine(`THRUNT God activated. Hunt root: ${huntRoot.fsPath}`);
}
```

The `findHuntRoot()` function must:
1. Iterate `vscode.workspace.workspaceFolders`
2. For each folder, check if `.hunt/MISSION.md` exists (use `vscode.workspace.fs.stat()`)
3. If not found, check `.planning/MISSION.md`
