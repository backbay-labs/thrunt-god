**Navigation strategy per CONTEXT.md decision:**
- Click sends a message to the background script which opens the artifact URL in a new tab (for vendor console links) or focuses a sidepanel detail view.
- Since the extension doesn't have artifact URLs stored directly, navigation sends a `command` message to the background script with a `navigate_artifact` command type. The background script can then use the bridge to resolve the artifact path.

**4. Extend bindActions()** to wire all click handlers:

After the existing button handlers, add:

```typescript
// Timeline item clicks (SIDE-06)
document.querySelectorAll('.timeline-item').forEach((el) => {
  el.addEventListener('click', () => {
    const artifactId = (el as HTMLElement).dataset.artifactId;
    const artifactType = (el as HTMLElement).dataset.artifactType;
    if (artifactId && artifactType) {
      chrome.runtime.sendMessage({
        type: 'navigate:artifact',
        artifactId,
        artifactType,
      });
      lastAction = `Navigating to ${artifactType} ${artifactId}`;
      lastActionTone = 'info';
      render();
      setTimeout(() => { lastAction = null; render(); }, 3000);
    }
  });
});

// Hypothesis clicks (SIDE-06)
document.querySelectorAll('.hypothesis-item').forEach((el) => {
  el.addEventListener('click', () => {
    const hypothesisId = (el as HTMLElement).dataset.hypothesisId;
    if (hypothesisId) {
      chrome.runtime.sendMessage({
        type: 'navigate:artifact',
        artifactId: hypothesisId,
        artifactType: 'hypothesis',
      });
      lastAction = `Navigating to hypothesis ${hypothesisId}`;
      lastActionTone = 'info';
      render();
      setTimeout(() => { lastAction = null; render(); }, 3000);
    }
  });
});

// Action clicks (SIDE-06)
document.querySelectorAll('.action-item').forEach((el) => {
  el.addEventListener('click', () => {
    const actionId = (el as HTMLElement).dataset.actionId;
    if (actionId) {
      chrome.runtime.sendMessage({
        type: 'navigate:action',
        actionId,
      });
      lastAction = `Executing action: ${actionId}`;
      lastActionTone = 'info';
      render();
      setTimeout(() => { lastAction = null; render(); }, 3000);
    }
  });
});
```
