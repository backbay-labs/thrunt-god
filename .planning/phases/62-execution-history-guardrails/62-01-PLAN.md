```typescript
}
```

**2. Create `apps/vscode/src/executionLogger.ts`** with:

**Constants:**

**ExecutionLogger class:**
```typescript
export class ExecutionLogger {
  private readonly historyPath: string;

  constructor(private readonly workspaceRoot: string) {
    this.historyPath = path.join(workspaceRoot, HISTORY_DIR, HISTORY_FILENAME);
  }
```

Methods:
- `append(entry: ExecutionEntry): void` -- reads existing JSON array from file (or empty array if missing/corrupt), prepends entry, prunes to maxEntries, writes back atomically (write to `.run-history.json.tmp` then `fs.renameSync` to final path).
- `getRecent(limit?: number): ExecutionEntry[]` -- reads file, returns up to `limit` entries (default: all). Returns empty array if file missing/corrupt.
- `prune(): void` -- reads file, trims to maxEntries, writes back. No-op if already within limit.
- `getMaxEntries(): number` -- reads `thruntGod.executionHistory.maxEntries` from `vscode.workspace.getConfiguration('thruntGod')`, falls back to `DEFAULT_MAX_ENTRIES`.
- `clear(): void` -- deletes the history file if it exists.

For atomic write, use:
```typescript
const tmpPath = this.historyPath + '.tmp';
fs.writeFileSync(tmpPath, JSON.stringify(entries, null, 2));
fs.renameSync(tmpPath, this.historyPath);
```

Wrap all file reads in try/catch returning empty array on failure. Wrap all file writes in try/catch logging to console.error on failure.

**confirmMutatingAction function (exported, standalone):**
```typescript
export async function confirmMutatingAction(
  actionName: string,
  environment: string | null,
): Promise<boolean> {
  const envLabel = environment ? `Target: ${environment}` : 'Target: local workspace';
  const result = await vscode.window.showWarningMessage(
    `"${actionName}" is a mutating action that will change state.\n\n${envLabel}\n\nProceed?`,
    { modal: true },
    'Proceed',
  );
  return result === 'Proceed';
}
```

**Helper to build ExecutionEntry from command deck execution:**
```typescript
export function buildCommandEntry(
  name: string,
  args: string[],
  stdout: string,
  stderr: string,
  exitCode: number | null,
  startedAt: number,
  status: 'success' | 'failure',

  mutating: boolean,
): ExecutionEntry {

    type: 'command',
    name,
    args,
    stdout,
    stderr,
    exitCode,
    startedAt,
    duration: Date.now() - startedAt,
    status,
    environment,
    mutating,
  };
}
```

**Helper to build ExecutionEntry from RunbookRunRecord:**
```typescript
export function buildRunbookEntry(
  record: import('../shared/runbook').RunbookRunRecord,
  environment: string | null,
): ExecutionEntry {
  const stdout = record.stepResults.map(r => r.output).join('\n');
  return {
    id: `EXE-${Date.now()}-${Math.random().toString(36).slice(2, 6)}`,
    type: 'runbook',
    name: record.runbookName,
    args: Object.entries(record.inputs).map(([k, v]) => `${k}=${v}`),
    stdout,
    stderr: '',
    exitCode: record.status === 'success' ? 0 : 1,
    startedAt: record.startTime,
    duration: record.durationMs,
    status: record.status === 'success' ? 'success' : record.status === 'aborted' ? 'aborted' : 'failure',
    environment,
    mutating: true, // runbooks are always mutating (they execute steps)
  };
}
```
