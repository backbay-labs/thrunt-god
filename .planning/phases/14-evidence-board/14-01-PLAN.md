```typescript
export interface EvidenceBoardNode {
  id: string;
  type: 'hypothesis' | 'receipt' | 'query';
  label: string;
  tier: number;           // 0 = hypothesis, 1 = receipt, 2 = query
  verdict?: string;
  confidence?: string;
  deviationScore?: number;
}

export interface EvidenceBoardEdge {
  source: string;
  target: string;
  relationship: 'supports' | 'contradicts' | 'context';
}

export interface EvidenceBoardMatrixCell {
  hypothesisId: string;
  receiptId: string;
  relationship: 'supports' | 'contradicts' | 'context' | 'absent';
  deviationScore: number | null;
}

export interface EvidenceBoardViewModel {
  nodes: EvidenceBoardNode[];
  edges: EvidenceBoardEdge[];
  matrixCells: EvidenceBoardMatrixCell[];
  hypothesisIds: string[];
  receiptIds: string[];
  blindSpots: string[];
}

export type HostToEvidenceBoardMessage =
  | { type: 'init'; viewModel: EvidenceBoardViewModel; isDark: boolean }
  | { type: 'update'; viewModel: EvidenceBoardViewModel }
  | { type: 'theme'; isDark: boolean };

export type EvidenceBoardToHostMessage =
  | { type: 'webview:ready' }
  | { type: 'node:select'; nodeId: string }
  | { type: 'node:open'; nodeId: string }
  | { type: 'mode:toggle'; mode: 'graph' | 'matrix' }
  | { type: 'hypothesis:focus'; hypothesisId: string | null }
  | { type: 'blur' };
```

    **Step 2: Add deriveEvidenceBoard() method to HuntDataStore**

    Add below the existing `deriveHuntOverview()` method. Import `EvidenceBoardViewModel`, `EvidenceBoardNode`, `EvidenceBoardEdge`, `EvidenceBoardMatrixCell` from `'../shared/evidence-board'`.

    Logic for `deriveEvidenceBoard(): EvidenceBoardViewModel`:

    1. Get all hypotheses from `this.getHunt()?.hypotheses`. If loaded, flatten `[...active, ...parked, ...disproved]`. For each hypothesis, create an `EvidenceBoardNode`:
       - `id`: hypothesis.id (e.g., "HYP-01")
       - `type`: 'hypothesis'
       - `label`: hypothesis.assertion (first 80 chars, ellipsis if longer)
       - `tier`: 0
       - `verdict`: hypothesis.status (e.g., "Supported", "Open")
       - `confidence`: hypothesis.confidence
       - `deviationScore`: undefined (hypotheses don't have deviation scores)

    2. Get all receipts from `this.getReceipts()`. For each loaded receipt, create a node:
       - `id`: receipt.receiptId
       - `type`: 'receipt'
       - `label`: receipt.claim (first 80 chars, ellipsis if longer)
       - `tier`: 1
       - `verdict`: receipt.claimStatus (e.g., "supports", "contradicts")
       - `confidence`: receipt.confidence
       - `deviationScore`: receipt.anomalyFrame?.deviationScore.totalScore ?? undefined

    3. Get all queries from `this.getQueries()`. For each loaded query, create a node:
       - `id`: query.queryId
       - `type`: 'query'
       - `label`: query.title (first 80 chars, ellipsis if longer)
       - `tier`: 2
       - `verdict`: undefined
       - `confidence`: undefined
       - `deviationScore`: undefined

    4. Build edges array:
       - For each receipt: for each `relatedHypotheses` ID, add edge `{ source: receiptId, target: hypothesisId, relationship: receipt.claimStatus as relationship }`. Map claimStatus: "supports" -> "supports", "contradicts" -> "contradicts", anything else (including "inconclusive") -> "context".
       - For each receipt: for each `relatedQueries` ID, add edge `{ source: queryId, target: receiptId, relationship: 'context' }`.

    5. Build matrixCells array:
       - Collect all hypothesis IDs (from step 1 nodes where type='hypothesis').
       - Collect all receipt IDs (from step 2 nodes where type='receipt').
       - For each hypothesis x receipt pair: look up if an edge exists between them. If yes, use the edge relationship and receipt's deviationScore. If no, use relationship='absent' and deviationScore=null.

    6. Extract blindSpots:
       - Call `this.getEvidenceReview()`. If loaded and `blindSpots` is a non-empty string, split by `\n`, trim each, filter empty, return the array.
       - Otherwise return `[]`.

    7. Return `{ nodes, edges, matrixCells, hypothesisIds, receiptIds, blindSpots }`.
