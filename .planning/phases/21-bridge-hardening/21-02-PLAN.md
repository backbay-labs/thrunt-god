```typescript
import { runThruntCommand } from './thrunt-tools.ts';
import type { Logger } from './logger.ts';

export interface SubprocessHealthState {
  available: boolean;
  lastProbeAt: string | null;
  lastProbeOk: boolean;
  consecutiveFailures: number;
}

export interface SubprocessHealthMonitor {
  isAvailable(): boolean;
  getState(): SubprocessHealthState;
  probe(): Promise<boolean>;
  startPeriodicProbe(intervalMs?: number): void;
  stop(): void;
}

export function createSubprocessHealthMonitor(opts: {
  projectRoot: string;
  toolsPath?: string | null;
  logger: Logger;
  probeTimeoutMs?: number;
}): SubprocessHealthMonitor
```

Implementation:
- Internal state: `available = false`, `lastProbeAt: string | null = null`, `lastProbeOk = false`, `consecutiveFailures = 0`
- `probe()`: calls `runThruntCommand(projectRoot, ['--version'], toolsPath, { timeoutMs: probeTimeoutMs ?? 5000, logger })`. If result.ok is true: set available=true, lastProbeOk=true, consecutiveFailures=0. If result.ok is false: increment consecutiveFailures, set lastProbeOk=false, set available=false if consecutiveFailures >= 2 (allow one transient failure). Log probe result via logger.info('subprocess', 'health probe', { available, consecutiveFailures }).
- `startPeriodicProbe(intervalMs = 60_000)`: calls probe() immediately, then setInterval(probe, intervalMs)
- `stop()`: clearInterval
- `isAvailable()`: returns `available`
- `getState()`: returns full state object
